Настройка UFW

ℹ️
Правила ALLOW должны стоять выше правил DENY. Иначе UFW заблокирует трафик до того как дойдёт до разрешения портов. Используем insert 1 чтобы гарантировать правильный порядок.
ufw --force reset && \
ufw allow 22/tcp && \
ufw allow 443/tcp && \
ufw allow 5223/tcp && \
ufw allow 8530/tcp && \
ufw --force enable && \
ufw status numbered
⚠️
Если после настройки баны UFW не работают как ожидается — проверяй порядок правил командой ufw status numbered. ALLOW для нужных портов должен стоять выше DENY Anywhere.

04Генерация секретов

🔑
Секрет MTG v2 (FakeTLS) состоит из трёх частей:
ee — префикс FakeTLS режима
16 случайных байт — уникальный ID
hex домена — маскировочный домен (cloudflare/apple/microsoft)
Пример: ee + 3bfb32756cc6363024b1cb9268af8524 + 7777772e6170706c652e636f6d
⚠️
Сохраните все три секрета — они понадобятся в конфигах и ссылках для подключения!
# mtg1 — cloudflare.com порт 443
RAND1=$(openssl rand -hex 16)
SECRET1="ee${RAND1}$(echo -n 'www.cloudflare.com' | xxd -p)"
echo "mtg1 (443  cloudflare): $SECRET1"

# mtg2 — apple.com порт 5223 (Apple Push Notification)
RAND2=$(openssl rand -hex 16)
SECRET2="ee${RAND2}$(echo -n 'www.apple.com' | xxd -p)"
echo "mtg2 (5223 apple):      $SECRET2"

# mtg3 — microsoft.com порт 8530 (Windows Server Update Services)
RAND3=$(openssl rand -hex 16)
SECRET3="ee${RAND3}$(echo -n 'www.microsoft.com' | xxd -p)"
echo "mtg3 (8530 microsoft):  $SECRET3"

08Скрипты автобана MTG

mtg-autoban.sh — бан при 500+ соединениях с одного IP

🤖
Whitelist: 91.108.x.x и 149.154.x.x — Telegram DC, никогда не банятся.
Запускается каждые 10 минут через cron. Лог: /var/log/mtg-autoban.log 
cat > /usr/local/bin/mtg-autoban.sh << 'EOF'
#!/bin/bash
THRESHOLD=500
PORTS="443 5223 8530"
WHITELIST="91.108. 149.154."
LOGFILE="/var/log/mtg-autoban.log"

for PORT in $PORTS; do
    ss -tn state established sport = :$PORT \
        | awk 'NR>1 {print $4}' \
        | cut -d: -f1 \
        | grep -v -E "$(echo $WHITELIST | tr ' ' '|')" \
        | sort | uniq -c \
        | while read COUNT IP; do
            if [ -z "$IP" ]; then continue; fi
            if [ "$COUNT" -gt "$THRESHOLD" ]; then
                if ! ufw status | grep -q "DENY.*$IP"; then
                    echo "$(date): Banning $IP ($COUNT conn on :$PORT)" | tee -a $LOGFILE
                    ufw deny from $IP
                    logger "mtg-autoban: banned $IP ($COUNT connections on :$PORT)"
                fi
            fi
        done
done
EOF

chmod +x /usr/local/bin/mtg-autoban.sh && \
echo '*/10 * * * * root /usr/local/bin/mtg-autoban.sh' > /etc/cron.d/mtg-autoban && \
echo "✅ Автобан установлен"

mtg-killbanned.sh — убивает сессии забаненных IP

💤
Закомментирован в кроне по умолчанию. Включить при необходимости — убрать # из файла cron.
cat > /usr/local/bin/mtg-killbanned.sh << 'EOF'
#!/bin/bash
BANNED=$(ufw status | grep 'DENY' | awk '{print $3}' | grep -v 'Anywhere')
for IP in $BANNED; do
    COUNT=$(ss -tn state established | grep "$IP" | wc -l)
    if [ "$COUNT" -gt 0 ]; then
        echo "$(date): Killing $COUNT connections from $IP"
        ss -K dst "$IP" 2>/dev/null
    fi
done
EOF

chmod +x /usr/local/bin/mtg-killbanned.sh && \
echo '#*/5 * * * * root /usr/local/bin/mtg-killbanned.sh' > /etc/cron.d/mtg-killbanned && \
echo "✅ Готово (выключен)"

#!/bin/bash
IFACE=$(ip route | grep default | awk '{print $5}' | head -1)
echo "🌐 СЕТЕВАЯ СТАТИСТИКА ($IFACE)"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "⚡ Текущая загрузка (5 сек замер):"
echo ""
RX1=$(cat /sys/class/net/$IFACE/statistics/rx_bytes)
TX1=$(cat /sys/class/net/$IFACE/statistics/tx_bytes)
sleep 5
RX2=$(cat /sys/class/net/$IFACE/statistics/rx_bytes)
TX2=$(cat /sys/class/net/$IFACE/statistics/tx_bytes)
RX_RATE=$(echo "scale=2; ($RX2 - $RX1) * 8 / 5 / 1000000" | bc)
TX_RATE=$(echo "scale=2; ($TX2 - $TX1) * 8 / 5 / 1000000" | bc)
echo "   ⬇️  RX: ${RX_RATE} Mbps"
echo "   ⬆️  TX: ${TX_RATE} Mbps"
sar_stats() {
    local lines=$1
    local data=$(sar -n DEV 2>/dev/null | grep "$IFACE" | tail -$lines)
    echo "$data" | awk '{s_rx+=$5;s_tx+=$6;if($5>m_rx)m_rx=$5;if($6>m_tx)m_tx=$6;c++} END{if(c>0)printf "%.2f %.2f %.2f %.2f\n",s_rx*8/1000/c,m_rx*8/1000,s_tx*8/1000/c,m_tx*8/1000;else print "0 0 0 0"}'
}
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "🕐 За 2 часа (avg / макс):"
echo ""

echo "   ⬆️  TX: avg ${AVG_TX:-0} Mbps / макс ${MAX_TX:-0} Mbps"
echo ""

echo ""

#!/bin/bash

        for IP in $BANNED; do
            INFO=$(curl -s https://ipinfo.io/$IP | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('org','?'), '|', d.get('city','?'))" 2>/dev/null)
            echo "   🔴 $IP — $INFO"
        done
    fi

#!/bin/bash
human_schedule() {
    local s=$1
    case "$s" in
        "*/10 * * * *") echo "каждые 10 минут" ;;
        "*/5 * * * *")  echo "каждые 5 минут" ;;
        "*/2 * * * *")  echo "каждые 2 минуты" ;;
        "59 23 * * *")  echo "ежедневно в 23:59" ;;
        *)              echo "$s" ;;
    esac
}
echo "⏰ CRON ЗАДАЧИ"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
for f in /etc/cron.d/mtg-autoban /etc/cron.d/mtg-killbanned /etc/cron.d/mtg-logger /etc/cron.d/sysstat; do
    [[ ! -f "$f" ]] && continue
    NAME=$(basename $f)
    echo ""
    echo "📋 $NAME:"
    while IFS= read -r line; do
        [[ "$line" =~ ^[[:space:]]*$ ]] && continue
        [[ "$line" =~ ^(PATH|SHELL|MAILTO)= ]] && continue
        if [[ "$line" =~ ^#[*/0-9] ]]; then
            REST="${line:1}"
            SCHEDULE=$(echo "$REST" | awk '{print $1,$2,$3,$4,$5}')
            CMD=$(echo "$REST" | awk '{print $NF}' | xargs basename)
            echo "   💤 $(human_schedule "$SCHEDULE") → $CMD"
            continue
        fi
        [[ "$line" =~ ^# ]] && continue
        SCHEDULE=$(echo "$line" | awk '{print $1,$2,$3,$4,$5}')
        CMD=$(echo "$line" | awk '{print $NF}' | xargs basename)
        echo "   🔹 $(human_schedule "$SCHEDULE") → $CMD"
    done < "$f"
done
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
EOF
