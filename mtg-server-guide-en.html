02Initial setup сервера

System update and install dependencies

apt update && apt upgrade -y && \
apt install -y ufw fail2ban curl wget bc sysstat python3

Set timezone

timedatectl set-timezone Europe/Moscow && date

UFW setup

ℹ️
ALLOW rules must be placed above DENY rules. Otherwise UFW will block traffic before reaching the allow rules. Use insert 1 to guarantee correct ordering.
ufw --force reset && \
ufw allow 22/tcp && \
ufw allow 443/tcp && \
ufw allow 5223/tcp && \
ufw allow 8530/tcp && \
ufw --force enable && \
ufw status numbered
⚠️
If UFW bans are not working as expected — check rule ordering with ufw status numbered. ALLOW for required ports must be above DENY Anywhere.

04Generate secrets

🔑
MTG v2 (FakeTLS) secret consists of three parts:
ee — FakeTLS mode prefix
16 random bytes — unique ID
domain hex — masquerade domain (cloudflare/apple/microsoft)
Пример: ee + 3bfb32756cc6363024b1cb9268af8524 + 7777772e6170706c652e636f6d
⚠️
Save all three secrets — you will need them in configs and connection links!
# mtg1 — cloudflare.com port 443
RAND1=$(openssl rand -hex 16)
SECRET1="ee${RAND1}$(echo -n 'www.cloudflare.com' | xxd -p)"
echo "mtg1 (443  cloudflare): $SECRET1"

# mtg2 — apple.com port 5223 (Apple Push Notification)
RAND2=$(openssl rand -hex 16)
SECRET2="ee${RAND2}$(echo -n 'www.apple.com' | xxd -p)"
echo "mtg2 (5223 apple):      $SECRET2"

# mtg3 — microsoft.com port 8530 (Windows Server Update Services)
RAND3=$(openssl rand -hex 16)
SECRET3="ee${RAND3}$(echo -n 'www.microsoft.com' | xxd -p)"
echo "mtg3 (8530 microsoft):  $SECRET3"

08MTG autoban scripts

mtg-autoban.sh — ban at 500+ connections from single IP

🤖
Whitelist: 91.108.x.x and 149.154.x.x — Telegram DC, never banned.
Runs every 10 minutes via cron. Log: /var/log/mtg-autoban.log 
cat > /usr/local/bin/mtg-autoban.sh << 'EOF'
#!/bin/bash
THRESHOLD=500
PORTS="443 5223 8530"
WHITELIST="91.108. 149.154."
LOGFILE="/var/log/mtg-autoban.log"

for PORT in $PORTS; do
    ss -tn state established sport = :$PORT \
        | awk 'NR>1 {print $4}' \
        | cut -d: -f1 \
        | grep -v -E "$(echo $WHITELIST | tr ' ' '|')" \
        | sort | uniq -c \
        | while read COUNT IP; do
            if [ -z "$IP" ]; then continue; fi
            if [ "$COUNT" -gt "$THRESHOLD" ]; then
                if ! ufw status | grep -q "DENY.*$IP"; then
                    echo "$(date): Banning $IP ($COUNT conn on :$PORT)" | tee -a $LOGFILE
                    ufw deny from $IP
                    logger "mtg-autoban: banned $IP ($COUNT connections on :$PORT)"
                fi
            fi
        done
done
EOF

chmod +x /usr/local/bin/mtg-autoban.sh && \
echo '*/10 * * * * root /usr/local/bin/mtg-autoban.sh' > /etc/cron.d/mtg-autoban && \
echo "✅ Autoban установлен"

mtg-killbanned.sh — kills sessions of banned IPs

💤
Disabled in cron by default. Enable when needed — remove # from the cron file.
cat > /usr/local/bin/mtg-killbanned.sh << 'EOF'
#!/bin/bash
BANNED=$(ufw status | grep 'DENY' | awk '{print $3}' | grep -v 'Anywhere')
for IP in $BANNED; do
    COUNT=$(ss -tn state established | grep "$IP" | wc -l)
    if [ "$COUNT" -gt 0 ]; then
        echo "$(date): Killing $COUNT connections from $IP"
        ss -K dst "$IP" 2>/dev/null
    fi
done
EOF

chmod +x /usr/local/bin/mtg-killbanned.sh && \
echo '#*/5 * * * * root /usr/local/bin/mtg-killbanned.sh' > /etc/cron.d/mtg-killbanned && \
echo "✅ Done (disabled)"

echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

#!/bin/bash
human_schedule() {
    local s=$1
    case "$s" in
        "*/10 * * * *") echo "every 10 minutes" ;;
        "*/5 * * * *")  echo "every 5 minutes" ;;
        "*/2 * * * *")  echo "every 2 minutes" ;;
        "59 23 * * *")  echo "daily at 23:59" ;;
        *)              echo "$s" ;;
    esac
}
echo "⏰ CRON JOBS"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
for f in /etc/cron.d/mtg-autoban /etc/cron.d/mtg-killbanned /etc/cron.d/mtg-logger /etc/cron.d/sysstat; do
    [[ ! -f "$f" ]] && continue
    NAME=$(basename $f)
    echo ""
    echo "📋 $NAME:"
    while IFS= read -r line; do
        [[ "$line" =~ ^[[:space:]]*$ ]] && continue
        [[ "$line" =~ ^(PATH|SHELL|MAILTO)= ]] && continue
        if [[ "$line" =~ ^#[*/0-9] ]]; then
            REST="${line:1}"
            SCHEDULE=$(echo "$REST" | awk '{print $1,$2,$3,$4,$5}')
            CMD=$(echo "$REST" | awk '{print $NF}' | xargs basename)
            echo "   💤 $(human_schedule "$SCHEDULE") → $CMD"
            continue
        fi
        [[ "$line" =~ ^# ]] && continue
        SCHEDULE=$(echo "$line" | awk '{print $1,$2,$3,$4,$5}')
        CMD=$(echo "$line" | awk '{print $NF}' | xargs basename)
        echo "   🔹 $(human_schedule "$SCHEDULE") → $CMD"
    done < "$f"
done
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
EOF
